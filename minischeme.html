/**************************************
 * The MiniScheme Grammar Starts Here *
 **************************************/
Goal ::= ( Line )* <EOF>
Line ::= Expression
Expression ::= IntegerLiteral
| TrueLiteral
| FalseLiteral
| Identifier
| BeginExpression
| PlusExpression
| MinusExpression
| MultiplyExpression
| DivideExpression
| GTExpression
| LTExpression
| EQExpression
| IfExpression
| DefineExpression
| Assignment
| ProcedureExp
| Application
IntegerLiteral ::= <INTEGER_LITERAL>
TrueLiteral ::= "#t"
FalseLiteral ::= "#f"
BeginExpression ::= "(" "begin" ( Expression )* ")"
PlusExpression ::= "(" "+" Expression Expression ")"
MinusExpression ::= "(" "-" Expression Expression ")"
MultiplyExpression ::= "(" "*" Expression Expression ")"
DivideExpression ::= "(" "/" Expression Expression ")"
GTExpression ::= "(" ">" Expression Expression ")"
LTExpression ::= "(" "<" Expression Expression ")"
EQExpression ::= "(" "=" Expression Expression ")"
IfExpression ::= "(" "if" Expression Expression Expression ")"
DefineExpression ::= "(" "define" Identifier Expression ")"
Identifier ::= <IDENTIFIER>
Assignment ::= "(" "set!" Identifier Expression ")"
ProcedureExp ::= "(" "lambda" "(" ( Identifier )* ")" Expression ")"
Application ::= "(" Expression ( Expression )* ")"
